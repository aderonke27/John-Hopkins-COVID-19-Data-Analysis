# Q12: visualize state/province wise active cases of novel coronavirus (COVID-19) in USA

#fill missing values to avoid errors in calculation
covid_data.fillna(0, inplace=True)

#filter for US
usa_df = covid_data[covid_data['Country/Region'] == 'US']

#group by state/pronvice and calculate totals
state_group = usa_df.groupby('Province/State')[['Confirmed', 'Deaths', 'Recovered']].sum()

#calculate active cases
state_group['Active'] = state_group['Confirmed'] - state_group['Deaths'] - state_group['Recovered']

#sort by active cases
state_group = state_group.sort_values(by='Active', ascending=False)

#plotting

plt.title('State/Province-wise active COVID-19 cases in USA')

# Q14: visualize worldwide confirmed novel coronavirus (COVID-19) cases over time

import os

#set the folder path
folder_path = r"C:\Users\DELL\Documents\COVID-19-master\COVID-19-master\csse_covid_19_data\csse_covid_19_daily_reports"

#read and combine all csv
all_data = []
for file in os.listdir(folder_path):
    if file.endswith('.csv'):
        file_path = os.path.join(folder_path, file)
        df = pd.read_csv(file_path)
        df['Date'] = pd.to_datetime(file.replace('.csv', ''), format='%m-%d-%Y')
        all_data.append(df)

#concatenate into one dataframe
combined_df = pd.concat(all_data, ignore_index=True)

#group by date and sum confirmed cases
worldwide_confirmed = (combined_df.groupby('Date')['Confirmed'].sum().sort_index())

#plot
plt.figure(figsize=(12, 6))
plt.plot(worldwide_confirmed.index, worldwide_confirmed.values, marker='1', color='blue')
plt.title('Worldwide Confirmed COVID-19 cases over time')
plt.xlabel('Date')
plt.ylabel('Total confirmed cases')
plt.grid(True)
plt.tight_layout()
plt.xticks(rotation=45)
plt.show()
